11  import net.sourceforge.pmd.PMD;

13  import org.junit.Test;

15  public class CSVRendererTest {
16      @Test
17      public void testLineCountPerFile() {
18          Renderer renderer = new CSVRenderer(true);
19          List<Match> list = new ArrayList<>();
20          String codeFragment = "code\nfragment";
21          Mark mark1 = createMark("public", "/var/Foo.java", 48, 10, codeFragment);
22          Mark mark2 = createMark("stuff", "/var/Bar.java", 73, 20, codeFragment);
23          Match match = new Match(75, mark1, mark2);
24  
25          list.add(match);
26          String report = renderer.render(list.iterator());
27          String expectedReport =
28              "tokens,occurrences" + PMD.EOL +
29              "75,2,48,10,/var/Foo.java,73,20,/var/Bar.java" + PMD.EOL;
30  
31          assertEquals(expectedReport, report);
32      }
33  
34      @Test
35      public void testFilenameEscapes() {
36          Renderer renderer = new CSVRenderer();
37          List<Match> list = new ArrayList<>();
38          String codeFragment = "code\nfragment";
39          Mark mark1 = createMark("public", "/var,with,commas/Foo.java", 48, 10, codeFragment);
40          Mark mark2 = createMark("stuff", "/var,with,commas/Bar.java", 73, 20, codeFragment);
41          Match match = new Match(75, mark1, mark2);
42          list.add(match);
43  
44          String report = renderer.render(list.iterator());
45          String expectedReport =
46                  "lines,tokens,occurrences" + PMD.EOL +
47                  "10,75,2,48,\"/var,with,commas/Foo.java\",73,\"/var,with,commas/Bar.java\"" + PMD.EOL;
48          assertEquals(expectedReport, report);
49      }
50  
51      private Mark createMark(String image, String tokenSrcID, int beginLine, int lineCount, String code) {
52          Mark result = new Mark(new TokenEntry(image, tokenSrcID, beginLine));
53  
54          result.setLineCount(lineCount);
55          result.setSoureCodeSlice(code);
56          return result;
57      }
58  }
