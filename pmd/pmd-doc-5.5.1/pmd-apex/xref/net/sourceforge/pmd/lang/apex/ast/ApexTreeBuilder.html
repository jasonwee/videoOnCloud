183 	private static <T extends AstNode> void register(Class<T> nodeType, Class<? extends ApexNode<T>> nodeAdapterType) {
184 		try {
185 			NODE_TYPE_TO_NODE_ADAPTER_TYPE.put(nodeType, nodeAdapterType.getConstructor(nodeType));
186 		} catch (SecurityException e) {
187 			throw new RuntimeException(e);
188 		} catch (NoSuchMethodException e) {
189 			throw new RuntimeException(e);
190 		}
191 	}

193 	// The nodes having children built.
194 	private Stack<Node> nodes = new Stack<>();
195 
196 	// The Apex nodes with children to build.
197 	private Stack<AstNode> parents = new Stack<>();
198 
199 	private SourceCodePositioner sourceCodePositioner;
200 
201 	public ApexTreeBuilder(String sourceCode) {
202 		sourceCodePositioner = new SourceCodePositioner(sourceCode);
203 	}
204 
205 	AdditionalPassScope scope = new AdditionalPassScope(new Errors());
206 
207 	static <T extends AstNode> ApexNode<T> createNodeAdapter(T node) {
208 		try {
209 			@SuppressWarnings("unchecked")
210 			// the register function makes sure only ApexNode<T> can be added,
211 			// where T is "T extends AstNode".
212 			Constructor<? extends ApexNode<T>> constructor = (Constructor<? extends ApexNode<T>>) NODE_TYPE_TO_NODE_ADAPTER_TYPE
213 					.get(node.getClass());
214 			if (constructor == null) {
215 				throw new IllegalArgumentException(
216 						"There is no Node adapter class registered for the Node class: " + node.getClass());
217 			}
218 			return constructor.newInstance(node);
219 		} catch (InstantiationException e) {
220 			throw new RuntimeException(e);
221 		} catch (IllegalAccessException e) {
222 			throw new RuntimeException(e);
223 		} catch (InvocationTargetException e) {
224 			throw new RuntimeException(e.getTargetException());
225 		}
226 	}
227 
228 	public <T extends AstNode> ApexNode<T> build(T astNode) {
229 		// Create a Node
230 		ApexNode<T> node = createNodeAdapter(astNode);
231 		calculateLineNumbers(node);
232 
233 		// Append to parent
234 		Node parent = nodes.isEmpty() ? null : nodes.peek();
235 		if (parent != null) {
236 			parent.jjtAddChild(node, parent.jjtGetNumChildren());
237 			node.jjtSetParent(parent);
238 		}
239 
240 		// Build the children...
241 		nodes.push(node);
242 		parents.push(astNode);
243 		astNode.traverse(this, scope);
244 		nodes.pop();
245 		parents.pop();
246 
247 		return node;
248 	}
249 
250 	private void calculateLineNumbers(ApexNode<?> node) {
251 		AbstractApexNode<?> apexNode = (AbstractApexNode<?>) node;
252 		apexNode.calculateLineNumbers(sourceCodePositioner);
253 	}
254 
255 	private boolean visit(AstNode node) {
256 		if (parents.peek() == node) {
257 			return true;
258 		} else {
259 			build(node);
260 			return false;
261 		}
262 	}
