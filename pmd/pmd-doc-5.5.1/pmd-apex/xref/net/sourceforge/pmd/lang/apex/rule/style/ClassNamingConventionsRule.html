4   package net.sourceforge.pmd.lang.apex.rule.style;
5   
6   import net.sourceforge.pmd.lang.apex.ast.ASTUserClass;
7   import net.sourceforge.pmd.lang.apex.ast.ASTUserInterface;
8   import net.sourceforge.pmd.lang.apex.rule.AbstractApexRule;
9   
10  public class ClassNamingConventionsRule extends AbstractApexRule {
11  	
12  	public ClassNamingConventionsRule() {
13  		setProperty(CODECLIMATE_CATEGORIES, new String[]{ "Style" });
14  		// Note: x10 as Apex has not automatic refactoring
15  		setProperty(CODECLIMATE_REMEDIATION_MULTIPLIER, 5);
16  		setProperty(CODECLIMATE_BLOCK_HIGHLIGHTING, false);
17  	}
18  
19  	public Object visit(ASTUserClass node, Object data) {
20  		if (Character.isLowerCase(node.getImage().charAt(0))) {
21  			addViolation(data, node);
22  		}
23  		return data;
24  	}
25  
26  	public Object visit(ASTUserInterface node, Object data) {
27  		if (Character.isLowerCase(node.getImage().charAt(0))) {
28  			addViolation(data, node);
29  		}
30  		return data;
31  	}
32  }
