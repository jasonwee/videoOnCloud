4   package net.sourceforge.pmd.lang.apex.ast;

6   import apex.jorje.semantic.ast.AstNode;
7   import net.sourceforge.pmd.lang.ast.RootNode;
8   import net.sourceforge.pmd.lang.ast.SourceCodePositioner;
9   
10  public abstract class ApexRootNode<T extends AstNode> extends AbstractApexNode<T> implements RootNode {
11  	public ApexRootNode(T node) {
12  		super(node);
13  	}
14  
15  	// For top level classes, the end is the end of file.
16  	@Override
17  	void calculateLineNumbers(SourceCodePositioner positioner) {
18  		super.calculateLineNumbers(positioner);
19  		this.endLine = positioner.getLastLine();
20  		this.endColumn = positioner.getLastLineColumn();
21  	}
22  }
