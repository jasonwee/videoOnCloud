27   *  fixes for CTORS, and mods to readChar().

110     private void ExpandBuff(boolean wrapAround)
111     {
112         char[] newbuffer = new char[bufsize + nextBufExpand];
113         int[] newbufline = new int[bufsize + nextBufExpand];
114         int[] newbufcolumn = new int[bufsize + nextBufExpand];
115 
116         try
117         {
118             if (wrapAround)
119             {
120                 System.arraycopy(buffer, tokenBegin, newbuffer, 0, bufsize - tokenBegin);
121                 System.arraycopy(buffer, 0, newbuffer,
122                         bufsize - tokenBegin, bufpos);
123                 buffer = newbuffer;
124 
125                 System.arraycopy(bufline, tokenBegin, newbufline, 0, bufsize - tokenBegin);
126                 System.arraycopy(bufline, 0, newbufline, bufsize - tokenBegin, bufpos);
127                 bufline = newbufline;
128 
129                 System.arraycopy(bufcolumn, tokenBegin, newbufcolumn, 0, bufsize - tokenBegin);
130                 System.arraycopy(bufcolumn, 0, newbufcolumn, bufsize - tokenBegin, bufpos);
131                 bufcolumn = newbufcolumn;
132 
133                 maxNextCharInd = (bufpos += bufsize - tokenBegin);
134             }
135             else
136             {
137                 System.arraycopy(buffer, tokenBegin, newbuffer, 0, bufsize - tokenBegin);
138                 buffer = newbuffer;
139 
140                 System.arraycopy(bufline, tokenBegin, newbufline, 0, bufsize - tokenBegin);
141                 bufline = newbufline;
142 
143                 System.arraycopy(bufcolumn, tokenBegin, newbufcolumn, 0, bufsize - tokenBegin);
144                 bufcolumn = newbufcolumn;
145 
146                 maxNextCharInd = (bufpos -= tokenBegin);
147             }
148         }
149         catch (Throwable t)
150         {
151             throw new Error(t.getMessage());
152         }
153 
154 
155         bufsize += nextBufExpand;
156         nextBufExpand = bufsize;
157         available = bufsize;
158         tokenBegin = 0;
159     }

161     private void FillBuff() throws java.io.IOException
162     {
163         if (maxNextCharInd == available)
164         {
165             if (available == bufsize)
166             {
167                 if (tokenBegin > nextBufExpand)
168                 {
169                     bufpos = maxNextCharInd = 0;
170                     available = tokenBegin;
171                 }
172                 else if (tokenBegin < 0)
173                 {
174                     bufpos = maxNextCharInd = 0;
175                 }
176                 else
177                 {
178                     ExpandBuff(false);
179                 }
180             }
181             else if (available > tokenBegin)
182             {
183                 available = bufsize;
184             }
185             else if ((tokenBegin - available) < nextBufExpand)
186             {
187                 ExpandBuff(true);
188             }
189             else
190             {
191                 available = tokenBegin;
192             }
193         }
194 
195         int i;
196         try 
197         {
198             i = inputStream.read(buffer, maxNextCharInd, available - maxNextCharInd);
199             if (i == -1)
200             {
201                 inputStream.close();
202                 throw new java.io.IOException();
203             }
204             else
205             {
206                 maxNextCharInd += i;
207             }
208             return;
209         }
210         catch(java.io.IOException e) 
211         {
212             --bufpos;
213             backup(0);
214             if (tokenBegin == -1)
215             {
216                 tokenBegin = bufpos;
217             }
218             throw e;
219         }
220     }

235     private void UpdateLineColumn(char c)
236     {
237         column++;
238 
239         if (prevCharIsLF)
240         {
241             prevCharIsLF = false;
242             line += (column = 1);
243         }
244         else if (prevCharIsCR)
245         {
246             prevCharIsCR = false;
247             if (c == '\n')
248             {
249                 prevCharIsLF = true;
250             }
251             else
252             {
253                 line += (column = 1);
254             }
255         }
256 
257         switch (c)
258         {
259         case '\r' :
260             prevCharIsCR = true;
261             break;
262         case '\n' :
263             prevCharIsLF = true;
264             break;
265         case '\t' :
266             column--;
267             column += 8 - (column & 07);
268             break;
269         default :
270             break;
271         }
272 
273         bufline[bufpos] = line;
274         bufcolumn[bufpos] = column;
275     }

277     /**
278      * @see org.apache.velocity.runtime.parser.CharStream#readChar()
279      */
280     @Override
281     public char readChar() throws java.io.IOException
282     {
283         if (inBuf > 0)
284         {
285             --inBuf;
286 
287             /*
288              *  was : return (char)((char)0xff & buffer[(bufpos == bufsize - 1) ? (bufpos = 0) : ++bufpos]);
289              */
290             return  buffer[(bufpos == bufsize - 1) ? (bufpos = 0) : ++bufpos];
291         }
292 
293         bufpos++;
294         if (bufpos >= maxNextCharInd)
295         {
296             FillBuff();
297         }
298 
299         /*
300          *  was : char c = (char)((char)0xff & buffer[bufpos]);
301          */
302         char c = buffer[bufpos];
303 
304         UpdateLineColumn(c);
305         return c;
306     }

366     /**
367      * @see org.apache.velocity.runtime.parser.CharStream#backup(int)
368      */
369     @Override
370     public void backup(int amount) 
371     {
372 
373         inBuf += amount;
374         bufpos -= amount;
375         if (bufpos < 0) {
376             bufpos += bufsize;
377         }
378     }

386     public void ReInit(java.io.Reader dstream, int startline,
387             int startcolumn, int buffersize)
388     {
389         inputStream = dstream;
390         line = startline;
391         column = startcolumn - 1;
392 
393         if (buffer == null || buffersize != buffer.length)
394         {
395             available = bufsize = nextBufExpand = buffersize;
396             buffer = new char[buffersize];
397             bufline = new int[buffersize];
398             bufcolumn = new int[buffersize];
399         }
400         prevCharIsLF = prevCharIsCR = false;
401         tokenBegin = inBuf = maxNextCharInd = 0;
402         bufpos = -1;
403     }

440     @Override
441     public String GetImage()
442     {
443         if (bufpos >= tokenBegin)
444         {
445             return new String(buffer, tokenBegin, bufpos - tokenBegin + 1);
446         }
447         else
448         {
449             return new String(buffer, tokenBegin, bufsize - tokenBegin) +
450             new String(buffer, 0, bufpos + 1);
451         }
452     }

454     /**
455      * @see org.apache.velocity.runtime.parser.CharStream#GetSuffix(int)
456      */
457     @Override
458     public char[] GetSuffix(int len)
459     {
460         char[] ret = new char[len];
461 
462         if ((bufpos + 1) >= len)
463         {
464             System.arraycopy(buffer, bufpos - len + 1, ret, 0, len);
465         }
466         else
467         {
468             System.arraycopy(buffer, bufsize - (len - bufpos - 1), ret, 0,
469                     len - bufpos - 1);
470             System.arraycopy(buffer, 0, ret, len - bufpos - 1, bufpos + 1);
471         }
472 
473         return ret;
474     }

487     /**
488      * Method to adjust line and column numbers for the start of a token.<BR>
489      * @param newLine
490      * @param newCol
491      */
492     public void adjustBeginLineColumn(int newLine, int newCol)
493     {
494         int start = tokenBegin;
495         int len;
496 
497         if (bufpos >= tokenBegin)
498         {
499             len = bufpos - tokenBegin + inBuf + 1;
500         }
501         else
502         {
503             len = bufsize - tokenBegin + bufpos + 1 + inBuf;
504         }
505 
506         int i = 0;
507         int j = 0;
508         int k = 0;
509         int nextColDiff = 0;
510         int columnDiff = 0;
511 
512         while (i < len &&
513                 bufline[j = start % bufsize] == bufline[k = ++start % bufsize])
514         {
515             bufline[j] = newLine;
516             nextColDiff = columnDiff + bufcolumn[k] - bufcolumn[j];
517             bufcolumn[j] = newCol + columnDiff;
518             columnDiff = nextColDiff;
519             i++;
520         }
521 
522         if (i < len)
523         {
524             bufline[j] = newLine++;
525             bufcolumn[j] = newCol + columnDiff;
526 
527             while (i++ < len)
528             {
529                 j = start % bufsize;
530                 start++;
531                 if (bufline[j] != bufline[start % bufsize]) {
532                     bufline[j] = newLine++;
533                 } else {
534                     bufline[j] = newLine;
535                 }
536             }
537         }
538 
539         line = bufline[j];
540         column = bufcolumn[j];
541     }
