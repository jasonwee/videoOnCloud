197     static Node getCompilationUnit(LanguageVersionHandler languageVersionHandler, String code) {
198         Parser parser = languageVersionHandler.getParser(languageVersionHandler.getDefaultParserOptions());
199         Node node = parser.parse(null, new StringReader(code));
200         languageVersionHandler.getSymbolFacade().start(node);
201         languageVersionHandler.getTypeResolutionFacade(Designer.class.getClassLoader()).start(node);
202         return node;
203     }

208 	private static LanguageVersion[] getSupportedLanguageVersions() {
209 		List<LanguageVersion> languageVersions = new ArrayList<>();
210 		for (LanguageVersion languageVersion : LanguageRegistry.findAllVersions()) {
211 			LanguageVersionHandler languageVersionHandler = languageVersion.getLanguageVersionHandler();
212 			if (languageVersionHandler != null) {
213 				Parser parser = languageVersionHandler.getParser(languageVersionHandler.getDefaultParserOptions());
214 				if (parser != null && parser.canParse()) {
215 					languageVersions.add(languageVersion);
216 				}
217 			}
218 		}
219 		return languageVersions.toArray(new LanguageVersion[languageVersions.size()]);
220 	}

253 	private class ExceptionNode implements TreeNode {
254 
255 		private Object item;
256 		private ExceptionNode[] kids;
257 
258 		public ExceptionNode(Object theItem) {
259 			item = theItem;
260 
261 			if (item instanceof ParseException) {
262 				createKids();
263 			}
264 		}
265 
266 		// each line in the error message becomes a separate tree node
267 		private void createKids() {
268 
269 			String message = ((ParseException) item).getMessage();
270 			String[] lines = StringUtil.substringsOf(message, PMD.EOL);
271 
272 			kids = new ExceptionNode[lines.length];
273 			for (int i = 0; i < lines.length; i++) {
274 				kids[i] = new ExceptionNode(lines[i]);
275 			}
276 		}
277 
278 		@Override
279         public int getChildCount() {
280 			return kids == null ? 0 : kids.length;
281 		}
282 
283 		@Override
284         public boolean getAllowsChildren() {
285 			return false;
286 		}
287 
288 		@Override
289         public boolean isLeaf() {
290 			return kids == null;
291 		}
292 
293 		@Override
294         public TreeNode getParent() {
295 			return null;
296 		}
297 
298 		@Override
299         public TreeNode getChildAt(int childIndex) {
300 			return kids[childIndex];
301 		}
302 
303 		public String label() {
304 			return item.toString();
305 		}
306 
307 		@Override
308         public Enumeration<TreeNode> children() {
309 			Enumeration<TreeNode> e = new Enumeration<TreeNode>() {
310 				int i = 0;
311 
312 				@Override
313                 public boolean hasMoreElements() {
314 					return kids != null && i < kids.length;
315 				}
316 
317 				@Override
318                 public ExceptionNode nextElement() {
319 					return kids[i++];
320 				}
321 			};
322 			return e;
323 		}
324 
325 		@Override
326         public int getIndex(TreeNode node) {
327 			for (int i = 0; i < kids.length; i++) {
328 				if (kids[i] == node) {
329 					return i;
330 				}
331 			}
332 			return -1;
333 		}
334 	}

336 	// Tree node that wraps the AST node for the tree widget and
337 	// any possible children they may have.
338 	private class ASTTreeNode implements TreeNode {
339 
340 		private Node node;
341 		private ASTTreeNode parent;
342 		private ASTTreeNode[] kids;
343 
344 		public ASTTreeNode(Node theNode) {
345 			node = theNode;
346 
347 			Node parent = node.jjtGetParent();
348 			if (parent != null) {
349 				this.parent = new ASTTreeNode(parent);
350 			}
351 		}
352 
353 		private ASTTreeNode(ASTTreeNode parent, Node theNode) {
354 			node = theNode;
355 			this.parent = parent;
356 		}
357 
358 		@Override
359         public int getChildCount() {
360 			return node.jjtGetNumChildren();
361 		}
362 
363 		@Override
364         public boolean getAllowsChildren() {
365 			return false;
366 		}
367 
368 		@Override
369         public boolean isLeaf() {
370 			return node.jjtGetNumChildren() == 0;
371 		}
372 
373 		@Override
374         public TreeNode getParent() {
375 			return parent;
376 		}
377 
378 		public Scope getScope() {
379 			if (node instanceof ScopedNode) {
380 				return ((ScopedNode) node).getScope();
381 			}
382 			return null;
383 		}
384 
385 		@Override
386         public Enumeration<TreeNode> children() {
387 
388 			if (getChildCount() > 0) {
389 				getChildAt(0); // force it to build kids
390 			}
391 
392 			Enumeration<TreeNode> e = new Enumeration<TreeNode>() {
393 				int i = 0;
394 
395 				@Override
396                 public boolean hasMoreElements() {
397 					return kids != null && i < kids.length;
398 				}
399 
400 				@Override
401                 public ASTTreeNode nextElement() {
402 					return kids[i++];
403 				}
404 			};
405 			return e;
406 		}
407 
408 		@Override
409         public TreeNode getChildAt(int childIndex) {
410 
411 			if (kids == null) {
412 				kids = new ASTTreeNode[node.jjtGetNumChildren()];
413 				for (int i = 0; i < kids.length; i++) {
414 					kids[i] = new ASTTreeNode(this.parent, node.jjtGetChild(i));
415 				}
416 			}
417 			return kids[childIndex];
418 		}
419 
420 		@Override
421         public int getIndex(TreeNode node) {
422 
423 			for (int i = 0; i < kids.length; i++) {
424 				if (kids[i] == node) {
425 					return i;
426 				}
427 			}
428 			return -1;
429 		}
430 
431 		public String label() {
432 			LanguageVersionHandler languageVersionHandler = getLanguageVersionHandler();
433 			StringWriter writer = new StringWriter();
434 			languageVersionHandler.getDumpFacade(writer, "", false).start(node);
435 			return writer.toString();
436 		}
437 
438 		public String getToolTipText() {
439 			String tooltip = "Line: " + node.getBeginLine() + " Column: " + node.getBeginColumn();
440 			tooltip += " " + label();
441 			return tooltip;
442 		}
443 
444 		public List<String> getAttributes() {
445 			List<String> result = new LinkedList<>();
446 			AttributeAxisIterator attributeAxisIterator = new AttributeAxisIterator(node);
447 			while (attributeAxisIterator.hasNext()) {
448 				Attribute attribute = attributeAxisIterator.next();
449 				result.add(attribute.getName() + "=" + attribute.getStringValue());
450 			}
451 			return result;
452 		}
453 	}

474 		@Override
475 		public String convertValueToText(Object value, boolean selected, boolean expanded, boolean leaf, int row,
476 				boolean hasFocus) {
477 			if (value == null) {
478 				return "";
479 			}
480 			if (value instanceof ASTTreeNode) {
481 				return ((ASTTreeNode) value).label();
482 			}
483 			if (value instanceof ExceptionNode) {
484 				return ((ExceptionNode) value).label();
485 			}
486 			return value.toString();
487 		}
488 
489 		@Override
490 		public String getToolTipText(MouseEvent e) {
491 			if (getRowForLocation(e.getX(), e.getY()) == -1) {
492 				return null;
493 			}
494 			TreePath curPath = getPathForLocation(e.getX(), e.getY());
495 			if (curPath.getLastPathComponent() instanceof ASTTreeNode) {
496 				return ((ASTTreeNode) curPath.getLastPathComponent()).getToolTipText();
497 			} else {
498 				return super.getToolTipText(e);
499 			}
500 		}
501 
502 		public void expandAll(boolean expand) {
503 			TreeNode root = (TreeNode) getModel().getRoot();
504 			expandAll(new TreePath(root), expand);
505 		}
506 
507 		private void expandAll(TreePath parent, boolean expand) {
508 			// Traverse children
509 			TreeNode node = (TreeNode) parent.getLastPathComponent();
510 			if (node.getChildCount() >= 0) {
511 				for (Enumeration<? extends TreeNode> e = node.children(); e.hasMoreElements();) {
512 					TreeNode n = e.nextElement();
513 					TreePath path = parent.pathByAddingChild(n);
514 					expandAll(path, expand);
515 				}
516 			}
517 
518 			if (expand) {
519 				expandPath(parent);
520 			} else {
521 				collapsePath(parent);
522 			}
523 		}
524 	}

582     			List<DFAGraphMethod> methods = dfaGraphRule.getMethods();

591 	private class XPathListener implements ActionListener {
592 		@Override
593         public void actionPerformed(ActionEvent ae) {
594 			xpathResults.clear();
595 			if (StringUtil.isEmpty(xpathQueryArea.getText())) {
596 				xpathResults.addElement("XPath query field is empty.");
597 				xpathResultList.repaint();
598 				codeEditorPane.requestFocus();
599 				return;
600 			}
601 			Node c = getCompilationUnit();
602 			try {
603 				XPathRule xpathRule = new XPathRule() {
604 					@Override
605 					public void addViolation(Object data, Node node, String arg) {
606 						xpathResults.addElement(node);
607 					}
608 				};
609 				xpathRule.setMessage("");
610 				xpathRule.setLanguage(getLanguageVersion().getLanguage());
611 				xpathRule.setXPath(xpathQueryArea.getText());
612 				xpathRule.setVersion(xpathVersionButtonGroup.getSelection().getActionCommand());
613 
614 				RuleSet ruleSet = new RuleSet();
615 				ruleSet.addRule(xpathRule);
616 
617 				RuleSets ruleSets = new RuleSets(ruleSet);
618 
619 				RuleContext ruleContext = new RuleContext();
620 				ruleContext.setLanguageVersion(getLanguageVersion());
621 
622 				List<Node> nodes = new ArrayList<>();
623 				nodes.add(c);
624 				ruleSets.apply(nodes, ruleContext, xpathRule.getLanguage());
625 
626 				if (xpathResults.isEmpty()) {
627 					xpathResults.addElement("No matching nodes " + System.currentTimeMillis());
628 				}
629 			} catch (ParseException pe) {
630 				xpathResults.addElement(pe.fillInStackTrace().getMessage());
631 			}
632 			xpathResultList.repaint();
633 			xpathQueryArea.requestFocus();
634 		}
635 	}

637 	private class SymbolTableListener implements TreeSelectionListener {
638 		@Override
639         public void valueChanged(TreeSelectionEvent e) {
640 			if (e.getNewLeadSelectionPath() != null) {
641 				ASTTreeNode astTreeNode = (ASTTreeNode) e.getNewLeadSelectionPath().getLastPathComponent();
642 
643 				DefaultMutableTreeNode symbolTableTreeNode = new DefaultMutableTreeNode();
644 				DefaultMutableTreeNode selectedAstTreeNode = new DefaultMutableTreeNode("AST Node: "
645 						+ astTreeNode.label());
646 				symbolTableTreeNode.add(selectedAstTreeNode);
647 
648 				List<Scope> scopes = new ArrayList<>();
649 				Scope scope = astTreeNode.getScope();
650 				while (scope != null) {
651 					scopes.add(scope);
652 					scope = scope.getParent();
653 				}
654 				Collections.reverse(scopes);
655 				for (int i = 0; i < scopes.size(); i++) {
656 					scope = scopes.get(i);
657 					DefaultMutableTreeNode scopeTreeNode = new DefaultMutableTreeNode("Scope: "
658 							+ scope.getClass().getSimpleName());
659 					selectedAstTreeNode.add(scopeTreeNode);
660 					for (Map.Entry<NameDeclaration, List<NameOccurrence>> entry : scope.getDeclarations().entrySet()) {
661 					    DefaultMutableTreeNode nameDeclarationTreeNode = new DefaultMutableTreeNode(
662 					            entry.getKey().getClass().getSimpleName() + ": " + entry.getKey());
663 					    scopeTreeNode.add(nameDeclarationTreeNode);
664 					    for (NameOccurrence nameOccurrence : entry.getValue()) {
665 					        DefaultMutableTreeNode nameOccurranceTreeNode = new DefaultMutableTreeNode(
666 					                "Name occurrence: " + nameOccurrence);
667 					        nameDeclarationTreeNode.add(nameOccurranceTreeNode);
668 					    }
669 					}
670 				}
671 
672 				List<String> attributes = astTreeNode.getAttributes();
673 				DefaultMutableTreeNode attributesNode = new DefaultMutableTreeNode("Attributes (accessible via XPath):");
674 				selectedAstTreeNode.add(attributesNode);
675 				for (String attribute : attributes) {
676 					attributesNode.add(new DefaultMutableTreeNode(attribute));
677 				}
678 
679 				loadSymbolTableTreeData(symbolTableTreeNode);
680 			}
681 		}
682 	}

894 	private static void makeTextComponentUndoable(JTextComponent textConponent) {
895 		final UndoManager undoManager = new UndoManager();
896 		textConponent.getDocument().addUndoableEditListener(new UndoableEditListener() {
897 			@Override
898             public void undoableEditHappened(UndoableEditEvent evt) {
899 				undoManager.addEdit(evt.getEdit());
900 			}
901 		});
902 		ActionMap actionMap = textConponent.getActionMap();
903 		InputMap inputMap = textConponent.getInputMap();
904 		actionMap.put("Undo", new AbstractAction("Undo") {
905 			@Override
906             public void actionPerformed(ActionEvent evt) {
907 				try {
908 					if (undoManager.canUndo()) {
909 						undoManager.undo();
910 					}
911 				} catch (CannotUndoException e) {
912 				}
913 			}
914 		});
915 		inputMap.put(KeyStroke.getKeyStroke("control Z"), "Undo");
916 
917 		actionMap.put("Redo", new AbstractAction("Redo") {
918 			@Override
919             public void actionPerformed(ActionEvent evt) {
920 				try {
921 					if (undoManager.canRedo()) {
922 						undoManager.redo();
923 					}
924 				} catch (CannotRedoException e) {
925 				}
926 			}
927 		});
928 		inputMap.put(KeyStroke.getKeyStroke("control Y"), "Redo");
929 	}

966 	/**
967 	 * Returns an unformatted xml string (without the declaration)
968 	 *
969 	 * @throws TransformerException if the XML cannot be converted to a string
970 	 */
971 	private static String getXmlString(Node node) throws TransformerException {
972 		StringWriter writer = new StringWriter();
973 
974 		Source source = new DOMSource(node.getAsDocument());
975 		Result result = new StreamResult(writer);
976 		TransformerFactory transformerFactory = TransformerFactory.newInstance();
977 		Transformer xformer = transformerFactory.newTransformer();
978 		xformer.setOutputProperty(OutputKeys.INDENT, "yes");
979 		xformer.transform(source, result);
980 
981 		return writer.toString();
982 	}

988 	private void loadSettings() {
989 	    InputStream stream = null;
990 		try {
991 			File file = new File(SETTINGS_FILE_NAME);
992 			if (file.exists()) {
993 				DocumentBuilder builder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
994 				stream = new FileInputStream(file);
995 				Document document = builder.parse(stream);
996 				Element settingsElement = document.getDocumentElement();
997 				Element codeElement = (Element) settingsElement.getElementsByTagName("code").item(0);
998 				Element xpathElement = (Element) settingsElement.getElementsByTagName("xpath").item(0);
999 
1000 				String code = getTextContext(codeElement);
1001 				String languageVersion = codeElement.getAttribute("language-version");
1002 				String xpath = getTextContext(xpathElement);
1003 				String xpathVersion = xpathElement.getAttribute("version");
1004 
1005 				codeEditorPane.setText(code);
1006 				setLanguageVersion(LanguageRegistry.findLanguageVersionByTerseName(languageVersion));
1007 				xpathQueryArea.setText(xpath);
1008 				for (Enumeration<AbstractButton> e = xpathVersionButtonGroup.getElements(); e.hasMoreElements();) {
1009 					AbstractButton button = e.nextElement();
1010 					if (xpathVersion.equals(button.getActionCommand())) {
1011 						button.setSelected(true);
1012 						break;
1013 					}
1014 				}
1015 			}
1016 		} catch (ParserConfigurationException e) {
1017 			e.printStackTrace();
1018 		} catch (IOException e) {
1019 			e.printStackTrace();
1020 		} catch (SAXException e) {
1021 			e.printStackTrace();
1022 		} finally {
1023 		    IOUtils.closeQuietly(stream);
1024 		}
1025 	}

1027 	private void saveSettings() {
1028 		try {
1029 			DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
1030 			DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder();
1031 			Document document = documentBuilder.newDocument();
1032 
1033 			Element settingsElement = document.createElement("settings");
1034 			document.appendChild(settingsElement);
1035 
1036 			Element codeElement = document.createElement("code");
1037 			settingsElement.appendChild(codeElement);
1038 			codeElement.setAttribute("language-version", getLanguageVersion().getTerseName());
1039 			codeElement.appendChild(document.createCDATASection(codeEditorPane.getText()));
1040 
1041 			Element xpathElement = document.createElement("xpath");
1042 			settingsElement.appendChild(xpathElement);
1043 			xpathElement.setAttribute("version", xpathVersionButtonGroup.getSelection().getActionCommand());
1044 			xpathElement.appendChild(document.createCDATASection(xpathQueryArea.getText()));
1045 
1046 			TransformerFactory transformerFactory = TransformerFactory.newInstance();
1047 			Transformer transformer = transformerFactory.newTransformer();
1048 			transformer.setOutputProperty(OutputKeys.METHOD, "xml");
1049 			// This is as close to pretty printing as we'll get using standard Java APIs.
1050 			transformer.setOutputProperty(OutputKeys.INDENT, "yes");
1051 			transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "3");
1052 			transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
1053 
1054 			Source source = new DOMSource(document);
1055 			Result result = new StreamResult(new FileWriter(new File(SETTINGS_FILE_NAME)));
1056 			transformer.transform(source, result);
1057 		} catch (ParserConfigurationException e) {
1058 			e.printStackTrace();
1059 		} catch (IOException e) {
1060 			e.printStackTrace();
1061 		} catch (TransformerException e) {
1062 			e.printStackTrace();
1063 		}
1064 	}

1066 	private String getTextContext(Element element) {
1067 		StringBuilder buf = new StringBuilder();
1068 		for (int i = 0; i < element.getChildNodes().getLength(); i++) {
1069 			org.w3c.dom.Node child = element.getChildNodes().item(i);
1070 			if (child instanceof Text) {
1071 				buf.append(((Text)child).getData());
1072 			}
1073 		}
1074 		return buf.toString();
1075 	}
