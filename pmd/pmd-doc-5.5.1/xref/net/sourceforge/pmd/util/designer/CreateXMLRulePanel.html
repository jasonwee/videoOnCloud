106     private static void appendLn(StringBuilder sb, String text) {
107         sb.append(text).append(PMD.EOL);
108     }
109 
110     /**
111      * We let our class implement the ActionListener interface
112      * and use it to generate the xml code when the user presses
113      * the "Create rule XML" button.
114      *
115      */
116     @Override
117     public void actionPerformed(ActionEvent exception) {
118     	
119     	boolean hasXPathQuery = StringUtil.isNotEmpty(xpathQueryArea.getText());
120     	
121         StringBuilder buffer = new StringBuilder(200);
122         appendLn(buffer, "<rule  name=\"" + rulenameField.getText() + '\"');
123         appendLn(buffer, "  message=\"" + rulemsgField.getText() + '\"');
124         appendLn(buffer, "  class=\"" + (hasXPathQuery ? "net.sourceforge.pmd.lang.rule.XPathRule" : "") + "\">");
125         appendLn(buffer, "  <description>");
126         appendLn(buffer, "  " + ruledescField.getText());
127         appendLn(buffer, "  </description>");
128         if (hasXPathQuery) {
129         	appendLn(buffer, "  <properties>");
130         	appendLn(buffer, "    <property name=\"xpath\">");
131         	appendLn(buffer, "    <value>");
132         	appendLn(buffer, "<![CDATA[");
133         	appendLn(buffer, xpathQueryArea.getText());
134         	appendLn(buffer, "]]>");
135         	appendLn(buffer, "    </value>");
136         	appendLn(buffer, "    </property>");
137         	appendLn(buffer, "  </properties>");
138         }
139         appendLn(buffer, "  <priority>3</priority>");
140         appendLn(buffer, "  <example>");
141         appendLn(buffer, "<![CDATA[");
142         appendLn(buffer, codeEditorPane.getText());
143         appendLn(buffer, "]]>");
144         appendLn(buffer, "  </example>");
145         appendLn(buffer, "</rule>");
146 
147         ruleXMLArea.setText(buffer.toString());
148         repaint();
149     }
