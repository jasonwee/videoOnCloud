4   package net.sourceforge.pmd.testframework;
5   
6   import java.util.ArrayList;
7   import java.util.List;
8   
9   import org.junit.runner.RunWith;
10  
11  import net.sourceforge.pmd.Rule;
12  
13  /**
14   * Standard methods for (simple) testcases.
15   */
16  @RunWith(PMDTestRunner.class)
17  public abstract class SimpleAggregatorTst extends RuleTst {
18  
19      private List<Rule> rules = new ArrayList<>();
20  
21      /**
22       * Configure the rule tests to be executed. Implement this method in
23       * subclasses by calling adRule.
24       * 
25       * @see #addRule(String, String)
26       */
27      protected void setUp() {
28          // empty, to be overridden
29      }
30  
31      /**
32       * Run a set of tests defined in an XML test-data file for a rule. The file
33       * should be ./xml/RuleName.xml relative to the test-class. The format is
34       * defined in test-data.xsd.
35       */
36      public void runTests(Rule rule) {
37          runTests(extractTestsFromXml(rule));
38      }
39  
40      /**
41       * Run a set of tests defined in a XML test-data file. The file should be
42       * ./xml/[testsFileName].xml relative to the test-class. The format is
43       * defined in test-data.xsd.
44       */
45      public void runTests(Rule rule, String testsFileName) {
46          runTests(extractTestsFromXml(rule, testsFileName));
47      }
48  
49      /**
50       * Run a set of tests of a certain sourceType.
51       */
52      public void runTests(TestDescriptor[] tests) {
53          for (int i = 0; i < tests.length; i++) {
54              runTest(tests[i]);
55          }
56      }
57  
58      /**
59       * Add new XML tests associated with the rule to the test suite. This should
60       * be called from the setup method.
61       */
62      protected void addRule(String ruleSet, String ruleName) {
63          rules.add(findRule(ruleSet, ruleName));
64      }
65  
66      /**
67       * Gets all configured rules.
68       * 
69       * @return all configured rules.
70       */
71      protected List<Rule> getRules() {
72          return rules;
73      }
74  }
