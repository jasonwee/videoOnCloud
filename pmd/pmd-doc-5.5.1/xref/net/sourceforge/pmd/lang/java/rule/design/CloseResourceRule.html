39  /**
40   * Makes sure you close your database connections. It does this by looking for
41   * code patterned like this:
42   * 
43   * <pre>
44   *  Connection c = X;
45   *  try {
46   *   // do stuff, and maybe catch something
47   *  } finally {
48   *   c.close();
49   *  }
50   * 
51   *  @author original author unknown
52   *  @author Contribution from Pierre Mathien
53   * </pre>
54   */
55  public class CloseResourceRule extends AbstractJavaRule {
56  
57      private Set<String> types = new HashSet<>();
58      private Set<String> simpleTypes = new HashSet<>();
59  
60      private Set<String> closeTargets = new HashSet<>();
61      private static final StringMultiProperty CLOSE_TARGETS_DESCRIPTOR = new StringMultiProperty("closeTargets",
62              "Methods which may close this resource", new String[] {}, 1.0f, ',');
63  
64      private static final StringMultiProperty TYPES_DESCRIPTOR = new StringMultiProperty("types", "Affected types",
65              new String[] { "java.sql.Connection", "java.sql.Statement", "java.sql.ResultSet" }, 2.0f, ',');
66  
67      private static final BooleanProperty USE_CLOSE_AS_DEFAULT_TARGET = new BooleanProperty("closeAsDefaultTarget",
68              "Consider 'close' as a target by default", true, 3.0f);
69  
70      public CloseResourceRule() {
71          definePropertyDescriptor(CLOSE_TARGETS_DESCRIPTOR);
72          definePropertyDescriptor(TYPES_DESCRIPTOR);
73          definePropertyDescriptor(USE_CLOSE_AS_DEFAULT_TARGET);
74      }
75  
76      @Override
77      public Object visit(ASTCompilationUnit node, Object data) {
78          if (closeTargets.isEmpty() && getProperty(CLOSE_TARGETS_DESCRIPTOR) != null) {
79              closeTargets.addAll(Arrays.asList(getProperty(CLOSE_TARGETS_DESCRIPTOR)));
80          }
81          if (getProperty(USE_CLOSE_AS_DEFAULT_TARGET) && !closeTargets.contains("close")) {
82              closeTargets.add("close");
83          }
84          if (types.isEmpty() && getProperty(TYPES_DESCRIPTOR) != null) {
85              types.addAll(Arrays.asList(getProperty(TYPES_DESCRIPTOR)));
86          }
87          if (simpleTypes.isEmpty() && getProperty(TYPES_DESCRIPTOR) != null) {
88              for (String type : getProperty(TYPES_DESCRIPTOR)) {
89                  simpleTypes.add(toSimpleType(type));
90              }
91          }
92          return super.visit(node, data);
93      }
94  
95      private static String toSimpleType(String fullyQualifiedClassName) {
96          int lastIndexOf = fullyQualifiedClassName.lastIndexOf('.');
97          if (lastIndexOf > -1) {
98              return fullyQualifiedClassName.substring(lastIndexOf + 1);
99          } else {
100             return fullyQualifiedClassName;
101         }
102     }

116     private void checkForResources(Node node, Object data) {
117         List<ASTLocalVariableDeclaration> vars = node.findDescendantsOfType(ASTLocalVariableDeclaration.class);
118         List<ASTVariableDeclaratorId> ids = new ArrayList<>();
119 
120         // find all variable references to Connection objects
121         for (ASTLocalVariableDeclaration var : vars) {
122             ASTType type = var.getTypeNode();
123 
124             if (type.jjtGetChild(0) instanceof ASTReferenceType) {
125                 ASTReferenceType ref = (ASTReferenceType) type.jjtGetChild(0);
126                 if (ref.jjtGetChild(0) instanceof ASTClassOrInterfaceType) {
127                     ASTClassOrInterfaceType clazz = (ASTClassOrInterfaceType) ref.jjtGetChild(0);
128 
129                     if (clazz.getType() != null && types.contains(clazz.getType().getName())
130                             || clazz.getType() == null && simpleTypes.contains(toSimpleType(clazz.getImage())) && !clazz.isReferenceToClassSameCompilationUnit()
131                             || types.contains(clazz.getImage()) && !clazz.isReferenceToClassSameCompilationUnit()) {
132 
133                         ASTVariableDeclaratorId id = var.getFirstDescendantOfType(ASTVariableDeclaratorId.class);
134                         ids.add(id);
135                     }
136                 }
137             }
138         }
139 
140         // if there are connections, ensure each is closed.
141         for (ASTVariableDeclaratorId x : ids) {
142             ensureClosed((ASTLocalVariableDeclaration) x.jjtGetParent().jjtGetParent(), x, data);
143         }
144     }

146     private boolean hasNullInitializer(ASTLocalVariableDeclaration var) {
147         ASTVariableInitializer init = var.getFirstDescendantOfType(ASTVariableInitializer.class);
148         if (init != null) {
149             try {
150                 List<?> nulls = init
151                         .findChildNodesWithXPath("Expression/PrimaryExpression/PrimaryPrefix/Literal/NullLiteral");
152                 return !nulls.isEmpty();
153             } catch (JaxenException e) {
154                 return false;
155             }
156         }
157         return false;
158     }

160     private void ensureClosed(ASTLocalVariableDeclaration var, ASTVariableDeclaratorId id, Object data) {
161         // What are the chances of a Connection being instantiated in a
162         // for-loop init block? Anyway, I'm lazy!
163         String variableToClose = id.getImage();
164         Node n = var;
165 
166         while (!(n instanceof ASTBlock) && !(n instanceof ASTConstructorDeclaration)) {
167             n = n.jjtGetParent();
168         }
169 
170         Node top = n;
171 
172         List<ASTTryStatement> tryblocks = top.findDescendantsOfType(ASTTryStatement.class);
173 
174         boolean closed = false;
175 
176         ASTBlockStatement parentBlock = id.getFirstParentOfType(ASTBlockStatement.class);
177 
178         // look for try blocks below the line the variable was
179         // introduced and make sure there is a .close call in a finally
180         // block.
181         for (ASTTryStatement t : tryblocks) {
182 
183             // verifies that there are no critical statements between the
184             // variable declaration and
185             // the beginning of the try block.
186             ASTBlockStatement tryBlock = t.getFirstParentOfType(ASTBlockStatement.class);
187             if (!hasNullInitializer(var) // no need to check for critical statements, if
188                                          // the variable has been initialized with null
189                 && parentBlock.jjtGetParent() == tryBlock.jjtGetParent()) {
190 
191                 List<ASTBlockStatement> blocks = parentBlock.jjtGetParent().findChildrenOfType(ASTBlockStatement.class);
192                 int parentBlockIndex = blocks.indexOf(parentBlock);
193                 int tryBlockIndex = blocks.indexOf(tryBlock);
194                 boolean criticalStatements = false;
195 
196                 for (int i = parentBlockIndex + 1; i < tryBlockIndex; i++) {
197                     // assume variable declarations are not critical
198                     ASTLocalVariableDeclaration varDecl = blocks.get(i).getFirstDescendantOfType(
199                             ASTLocalVariableDeclaration.class);
200                     if (varDecl == null) {
201                         criticalStatements = true;
202                         break;
203                     }
204                 }
205                 if (criticalStatements) {
206                     break;
207                 }
208             }
209 
210             if (t.getBeginLine() > id.getBeginLine() && t.hasFinally()) {
211                 ASTBlock f = (ASTBlock) t.getFinally().jjtGetChild(0);
212                 List<ASTName> names = f.findDescendantsOfType(ASTName.class);
213                 for (ASTName oName : names) {
214                     String name = oName.getImage();
215                     if (name != null && name.contains(".")) {
216                         String[] parts = name.split("\\.");
217                         if (parts.length == 2) {
218                             String methodName = parts[1];
219                             String varName = parts[0];
220                             if (varName.equals(variableToClose) && closeTargets.contains(methodName)
221                                     && nullCheckIfCondition(f, oName, varName)) {
222                                 closed = true;
223                                 break;
224                             }
225 
226                         }
227                     }
228                 }
229                 if (closed) {
230                     break;
231                 }
232 
233                 List<ASTStatementExpression> exprs = new ArrayList<>();
234                 f.findDescendantsOfType(ASTStatementExpression.class, exprs, true);
235                 for (ASTStatementExpression stmt : exprs) {
236                     ASTPrimaryExpression expr = stmt.getFirstChildOfType(ASTPrimaryExpression.class);
237                     if (expr != null) {
238                         ASTPrimaryPrefix prefix = expr.getFirstChildOfType(ASTPrimaryPrefix.class);
239                         ASTPrimarySuffix suffix = expr.getFirstChildOfType(ASTPrimarySuffix.class);
240                         if (prefix != null && suffix != null) {
241                             if (prefix.getImage() == null) {
242                                 ASTName prefixName = prefix.getFirstChildOfType(ASTName.class);
243                                 if (prefixName != null && closeTargets.contains(prefixName.getImage())) {
244                                     // Found a call to a "close target" that is
245                                     // a direct
246                                     // method call without a "ClassName."
247                                     // prefix.
248                                     closed = variableIsPassedToMethod(expr, variableToClose);
249                                     if (closed) {
250                                         break;
251                                     }
252                                 }
253                             } else if (suffix.getImage() != null) {
254                                 String prefixPlusSuffix = prefix.getImage() + "." + suffix.getImage();
255                                 if (closeTargets.contains(prefixPlusSuffix)) {
256                                     // Found a call to a "close target" that is
257                                     // a method call
258                                     // in the form "ClassName.methodName".
259                                     closed = variableIsPassedToMethod(expr, variableToClose);
260                                     if (closed) {
261                                         break;
262                                     }
263                                 }
264                             }
265                             // look for primary suffix containing the close
266                             // Targets elements.
267                             // If the .close is executed in another class
268                             // accessed by a method
269                             // this form :
270                             // getProviderInstance().closeConnexion(connexion)
271                             // For this use case, we assume the variable is
272                             // correctly closed
273                             // in the other class since there is no way to
274                             // really check it.
275                             if (!closed) {
276                                 List<ASTPrimarySuffix> suffixes = new ArrayList<>();
277                                 expr.findDescendantsOfType(ASTPrimarySuffix.class, suffixes, true);
278                                 for (ASTPrimarySuffix oSuffix : suffixes) {
279                                     String suff = oSuffix.getImage();
280                                     if (closeTargets.contains(suff)) {
281                                         closed = variableIsPassedToMethod(expr, variableToClose);
282                                         if (closed) {
283                                             break;
284                                         }
285                                     }
286 
287                                 }
288                             }
289                         }
290                     }
291                 }
292                 if (closed) {
293                     break;
294                 }
295             }
296         }
297 
298         if (!closed) {
299             // See if the variable is returned by the method, which means the
300             // method is a utility for creating the db resource, which means of
301             // course it can't be closed by the method, so it isn't an error.
302             List<ASTReturnStatement> returns = new ArrayList<>();
303             top.findDescendantsOfType(ASTReturnStatement.class, returns, true);
304             for (ASTReturnStatement returnStatement : returns) {
305                 ASTName name = returnStatement.getFirstDescendantOfType(ASTName.class);
306                 if (name != null && name.getImage().equals(variableToClose)) {
307                     closed = true;
308                     break;
309                 }
310             }
311         }
312 
313         // if all is not well, complain
314         if (!closed) {
315             ASTType type = var.getFirstChildOfType(ASTType.class);
316             ASTReferenceType ref = (ASTReferenceType) type.jjtGetChild(0);
317             ASTClassOrInterfaceType clazz = (ASTClassOrInterfaceType) ref.jjtGetChild(0);
318             addViolation(data, id, clazz.getImage());
319         }
320     }

322     private boolean variableIsPassedToMethod(ASTPrimaryExpression expr, String variable) {
323         List<ASTName> methodParams = new ArrayList<>();
324         expr.findDescendantsOfType(ASTName.class, methodParams, true);
325         for (ASTName pName : methodParams) {
326             String paramName = pName.getImage();
327             // also check if we've got the a parameter (i.e if it's an argument
328             // !)
329             ASTArgumentList parentParam = pName.getFirstParentOfType(ASTArgumentList.class);
330             if (paramName.equals(variable) && parentParam != null) {
331                 return true;
332             }
333         }
334         return false;
335     }

337     private ASTIfStatement findIfStatement(ASTBlock enclosingBlock, Node node) {
338         ASTIfStatement ifStatement = node.getFirstParentOfType(ASTIfStatement.class);
339         List<ASTIfStatement> allIfStatements = enclosingBlock.findDescendantsOfType(ASTIfStatement.class);
340         if (ifStatement != null && allIfStatements.contains(ifStatement)) {
341             return ifStatement;
342         }
343         return null;
344     }

346     /**
347      * Checks, whether the given node is inside a if condition, and if so,
348      * whether this is a null check for the given varName.
349      *
350      * @param enclosingBlock
351      *            where to search for if statements
352      * @param node
353      *            the node, where the call for the close is done
354      * @param varName
355      *            the variable, that is maybe null-checked
356      * @return <code>true</code> if no if condition is involved or if the if
357      *         condition is a null-check.
358      */
359     private boolean nullCheckIfCondition(ASTBlock enclosingBlock, Node node, String varName) {
360         ASTIfStatement ifStatement = findIfStatement(enclosingBlock, node);
361         if (ifStatement != null) {
362             try {
363                 // find expressions like: varName != null or null != varName
364                 List<?> nodes = ifStatement.findChildNodesWithXPath("Expression/EqualityExpression[@Image='!=']"
365                         + "  [PrimaryExpression/PrimaryPrefix/Name[@Image='" + varName + "']]"
366                         + "  [PrimaryExpression/PrimaryPrefix/Literal/NullLiteral]");
367                 return !nodes.isEmpty();
368             } catch (JaxenException e) {
369                 // no boolean literals or other condition
370             }
371         }
372         return true;
373     }
