145 	@Override
146 	public Object visit(ASTCompilationUnit node, Object data) {
147 		String className = null;
148 		try {
149 			importedOnDemand = new ArrayList<>();
150 			importedClasses = new HashMap<>();
151 			className = getClassName(node);
152 			if (className != null) {
153 				populateClassName(node, className);
154 			}
155 		} catch (ClassNotFoundException e) {
156 			if (LOG.isLoggable(Level.FINE)) {
157 				LOG.log(Level.FINE, "Could not find class " + className + ", due to: " + e);
158 			}
159 		} catch (NoClassDefFoundError e) {
160 			if (LOG.isLoggable(Level.FINE)) {
161  				LOG.log(Level.FINE, "Could not find class " + className + ", due to: " + e);
162  			}
163 		} catch (LinkageError e) {
164 			if (LOG.isLoggable(Level.WARNING)) {
165 				LOG.log(Level.WARNING, "Could not find class " + className + ", due to: " + e);
166 			}
167 		} finally {
168 			populateImports(node);
169 		}
170 		return super.visit(node, data);
171 	}

184 		}
185 		return data;

197 		String typeName = node.getImage();
198 		if (node.jjtGetParent().hasDescendantOfType(ASTClassOrInterfaceBody.class)) {
199 			anonymousClassCounter++;
200 		    AbstractNode parent = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
201 		    if (parent == null) {
202 		        parent = node.getFirstParentOfType(ASTEnumDeclaration.class);
203 		    }
204             typeName = parent.getImage() + "$" + anonymousClassCounter;
205 		}
206 		populateType(node, typeName);
207 		return data;
208 	}

228 	@Override
229 	public Object visit(ASTName node, Object data) {
230 		/*
231 		 * Only doing this for nodes where getNameDeclaration is null this means
232 		 * it's not a named node, i.e. Static reference or Annotation Doing this
233 		 * for memory - TODO: Investigate if there is a valid memory concern or
234 		 * not
235 		 */
236 		if (node.getNameDeclaration() == null) {
237 			// Skip these scenarios as there is no type to populate in these cases:
238 			// 1) Parent is a PackageDeclaration, which is not a type
239 			// 2) Parent is a ImportDeclaration, this is handled elsewhere.
240 			if (!(node.jjtGetParent() instanceof ASTPackageDeclaration || node.jjtGetParent() instanceof ASTImportDeclaration)) {
241 				String name = node.getImage();
242 				if (name.indexOf('.') != -1) {
243 					name = name.substring(0, name.indexOf('.'));
244 				}
245 				populateType(node, name);
246 			}
247 		} else {
248 			// Carry over the type from the declaration
249 			if (node.getNameDeclaration().getNode() instanceof TypeNode) {
250 				node.setType(((TypeNode)node.getNameDeclaration().getNode()).getType());
251 			}
252 		}
253 		return super.visit(node, data);
254 	}

563 	// Roll up the type based on type of the first child node.
564 	private void rollupTypeUnary(TypeNode typeNode) {
565 		Node node = typeNode;
566 		if (node.jjtGetNumChildren() >= 1) {
567 			Node child = node.jjtGetChild(0);
568 			if (child instanceof TypeNode) {
569 				typeNode.setType(((TypeNode)child).getType());
570 			}
571 		}
572 	}

574 	// Roll up the type based on type of the first child node using Unary Numeric Promotion per JLS 5.6.1
575 	private void rollupTypeUnaryNumericPromotion(TypeNode typeNode) {
576 		Node node = typeNode;
577 		if (node.jjtGetNumChildren() >= 1) {
578 			Node child = node.jjtGetChild(0);
579 			if (child instanceof TypeNode) {
580 				Class<?> type = ((TypeNode)child).getType();
581 				if (type != null) {
582 					if ("byte".equals(type.getName()) || "short".equals(type.getName())
583 							|| "char".equals(type.getName())) {
584 						populateType(typeNode, "int");
585 					} else {
586 						typeNode.setType(((TypeNode)child).getType());
587 					}
588 				}
589 			}
590 		}
591 	}

593 	// Roll up the type based on type of the first and second child nodes using Binary Numeric Promotion per JLS 5.6.2
594 	private void rollupTypeBinaryNumericPromotion(TypeNode typeNode) {
595 		Node node = typeNode;
596 		if (node.jjtGetNumChildren() >= 2) {
597 			Node child1 = node.jjtGetChild(0);
598 			Node child2 = node.jjtGetChild(1);
599 			if (child1 instanceof TypeNode && child2 instanceof TypeNode) {
600 				Class<?> type1 = ((TypeNode)child1).getType();
601 				Class<?> type2 = ((TypeNode)child2).getType();
602 				if (type1 != null && type2 != null) {
603 					// Yeah, String is not numeric, but easiest place to handle it, only affects ASTAdditiveExpression
604 					if ("java.lang.String".equals(type1.getName()) || "java.lang.String".equals(type2.getName())) {
605 						populateType(typeNode, "java.lang.String");
606 					} else if ("boolean".equals(type1.getName()) || "boolean".equals(type2.getName())) {
607 						populateType(typeNode, "boolean");
608 					} else if ("double".equals(type1.getName()) || "double".equals(type2.getName())) {
609 						populateType(typeNode, "double");
610 					} else if ("float".equals(type1.getName()) || "float".equals(type2.getName())) {
611 						populateType(typeNode, "float");
612 					} else if ("long".equals(type1.getName()) || "long".equals(type2.getName())) {
613 						populateType(typeNode, "long");
614 					} else {
615 						populateType(typeNode, "int");
616 					}
617 				} else if (type1 != null || type2 != null) {
618 					// If one side is known to be a String, then the result is a String
619 					// Yeah, String is not numeric, but easiest place to handle it, only affects ASTAdditiveExpression
620 					if (type1 != null && "java.lang.String".equals(type1.getName())
621 							|| type2 != null && "java.lang.String".equals(type2.getName())) {
622 						populateType(typeNode, "java.lang.String");
623 					}
624 				}
625 			}
626 		}
627 	}
628 
629 	private void populateType(TypeNode node, String className) {
630 
631 		String qualifiedName = className;
632 		Class<?> myType = PRIMITIVE_TYPES.get(className);
633 		if (myType == null && importedClasses != null) {
634 			if (importedClasses.containsKey(className)) {
635 				qualifiedName = importedClasses.get(className);
636 			} else if (importedClasses.containsValue(className)) {
637 				qualifiedName = className;
638 			}
639 			if (qualifiedName != null) {
640 				try {
641 					/*
642 					 * TODO - the map right now contains just class names. if we
643 					 * use a map of classname/class then we don't have to hit
644 					 * the class loader for every type - much faster
645 					 */
646 					myType = pmdClassLoader.loadClass(qualifiedName);
647 				} catch (ClassNotFoundException e) {
648 					myType = processOnDemand(qualifiedName);
649 				} catch (NoClassDefFoundError e) {
650 					myType = processOnDemand(qualifiedName);
651 				} catch (LinkageError e) {
652 					myType = processOnDemand(qualifiedName);
653 				}
654 			}
655 		}
656 		if (myType == null && qualifiedName != null && qualifiedName.contains(".")) {
657 		    // try if the last part defines a inner class
658 		    String qualifiedNameInner = qualifiedName.substring(0, qualifiedName.lastIndexOf('.'))
659 		            + "$" + qualifiedName.substring(qualifiedName.lastIndexOf('.') + 1);
660 		    try {
661 		        myType = pmdClassLoader.loadClass(qualifiedNameInner);
662 		    } catch (Exception e) {
663 		        // ignored
664 		    }
665 		}
666 		if (myType == null && qualifiedName != null && !qualifiedName.contains(".")) {
667 		    // try again with java.lang....
668 		    try {
669 		        myType = pmdClassLoader.loadClass("java.lang." + qualifiedName);
670 		    } catch (Exception e) {
671 		        // ignored
672 		    }
673 		}
674 		if (myType != null) {
675 			node.setType(myType);
676 		}
677 	}

701 	private Class<?> processOnDemand(String qualifiedName) {
702 		for (String entry : importedOnDemand) {
703 			try {
704 				return pmdClassLoader.loadClass(entry + "." + qualifiedName);
705 			} catch (Throwable e) {
706 			}
707 		}
708 		return null;

711 	private String getClassName(ASTCompilationUnit node) {
712 		ASTClassOrInterfaceDeclaration classDecl = node.getFirstDescendantOfType(ASTClassOrInterfaceDeclaration.class);
713 		if (classDecl == null) {
714 			return null; // Happens if this compilation unit only contains an enum
715 		}
716 		if (node.declarationsAreInDefaultPackage()) {
717 			return classDecl.getImage();
718 		}
719 		ASTPackageDeclaration pkgDecl = node.getPackageDeclaration();
720 		importedOnDemand.add(pkgDecl.getPackageNameImage());
721 		return pkgDecl.getPackageNameImage() + "." + classDecl.getImage();
722 	}

724 	/**
725 	 * If the outer class wasn't found then we'll get in here
726 	 *
727 	 * @param node
728 	 */
729 	private void populateImports(ASTCompilationUnit node) {
730 		List<ASTImportDeclaration> theImportDeclarations = node.findChildrenOfType(ASTImportDeclaration.class);
731 
732 		importedClasses.putAll(JAVA_LANG);
733 
734 		// go through the imports
735 		for (ASTImportDeclaration anImportDeclaration : theImportDeclarations) {
736 			String strPackage = anImportDeclaration.getPackageName();
737 			if (anImportDeclaration.isImportOnDemand()) {
738 				importedOnDemand.add(strPackage);
739 			} else if (!anImportDeclaration.isImportOnDemand()) {
740 				String strName = anImportDeclaration.getImportedName();
741 				importedClasses.put(strName, strName);
742 				importedClasses.put(strName.substring(strPackage.length() + 1), strName);
743 			}
744 		}
745 	}

747 	private void populateClassName(ASTCompilationUnit node, String className) throws ClassNotFoundException {
748 		node.setType(pmdClassLoader.loadClass(className));
749 		importedClasses.putAll(pmdClassLoader.getImportedClasses(className));
750 	}
