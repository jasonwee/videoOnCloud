4   package net.sourceforge.pmd.lang.jsp.ast;

6   /**
7    * @author Pieter_Van_Raemdonck
8    * @since Created on 11-jan-2006
9    */
10  public class StartAndEndTagMismatchException extends SyntaxErrorException {
11  
12      private static final long serialVersionUID = 5434485938487458692L;
13  
14      public static final String START_END_TAG_MISMATCH_RULE_NAME
15              = "Start and End Tags of an XML Element must match.";
16  
17      private final int startLine;
18      private final int endLine;
19      private final int startColumn;
20      private final int endColumn;
21      private final String startTagName;
22      private final String endTagName;
23  
24      /**
25       * Public constructor.
26       *
27       * @param startLine
28       * @param startColumn
29       * @param startTagName
30       * @param endLine
31       * @param endColumn
32       * @param endTagName
33       */
34      public StartAndEndTagMismatchException(int startLine, int startColumn, String startTagName,
35                                             int endLine, int endColumn, String endTagName) {
36          super(endLine, START_END_TAG_MISMATCH_RULE_NAME);
37          this.startLine = startLine;
38          this.startColumn = startColumn;
39          this.startTagName = startTagName;
40  
41          this.endLine = endLine;
42          this.endColumn = endColumn;
43          this.endTagName = endTagName;
44      }
45  
46  
47      /**
48       * @return Returns the endColumn.
49       */
50      public int getEndColumn() {
51          return endColumn;
52      }
53  
54      /**
55       * @return Returns the endLine.
56       */
57      public int getEndLine() {
58          return endLine;
59      }
60  
61      /**
62       * @return Returns the startColumn.
63       */
64      public int getStartColumn() {
65          return startColumn;
66      }
67  
68      /**
69       * @return Returns the startLine.
70       */
71      public int getStartLine() {
72          return startLine;
73      }
74  
75      /* (non-Javadoc)
76       * @see java.lang.Throwable#getMessage()
77       */
78      @Override
79      public String getMessage() {
80          return "The start-tag of element \"" + startTagName + "\" (line "
81                  + startLine + ", column " + startColumn
82                  + ") does not correspond to the end-tag found: \""
83                  + endTagName + "\" (line " + endLine
84                  + ", column " + endColumn + ").";
85      }
86  }
