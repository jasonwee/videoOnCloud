79      public static String stringFromType(int intype) {
80          if (-1 == intype) {
81              return "<ROOT>";
82          }
83          if (!TYPE_MAP.containsKey(intype)) {
84              throw new RuntimeException("Couldn't find NodeType type id " + intype);
85          }
86          return TYPE_MAP.get(intype);
87      }
