4   package net.sourceforge.pmd.lang.apex.ast;
5   
6   import apex.jorje.semantic.ast.modifier.ModifierNode;
7   
8   public class ASTModifierNode extends AbstractApexNode<ModifierNode> implements AccessNode {
9   
10  	public ASTModifierNode(ModifierNode modifierNode) {
11  		super(modifierNode);
12  	}
13  
14  	public Object jjtAccept(ApexParserVisitor visitor, Object data) {
15  		return visitor.visit(this, data);
16  	}
17  
18  	@Override
19  	public int getModifiers() {
20  		return node.getModifiers().getJavaModifiers();
21  	}
22  
23  	@Override
24  	public boolean isPublic() {
25  		return (node.getModifiers().getJavaModifiers() & PUBLIC) == PUBLIC;
26  	}
27  
28  	@Override
29  	public boolean isProtected() {
30  		return (node.getModifiers().getJavaModifiers() & PROTECTED) == PROTECTED;
31  	}
32  
33  	@Override
34  	public boolean isPrivate() {
35  		return (node.getModifiers().getJavaModifiers() & PRIVATE) == PRIVATE;
36  	}
37  
38  	@Override
39  	public boolean isAbstract() {
40  		return (node.getModifiers().getJavaModifiers() & ABSTRACT) == ABSTRACT;
41  	}
42  
43  	@Override
44  	public boolean isStatic() {
45  		return (node.getModifiers().getJavaModifiers() & STATIC) == STATIC;
46  	}
47  
48  	@Override
49  	public boolean isFinal() {
50  		return (node.getModifiers().getJavaModifiers() & FINAL) == FINAL;
51  	}
52  
53  	@Override
54  	public boolean isTransient() {
55  		return (node.getModifiers().getJavaModifiers() & TRANSIENT) == TRANSIENT;
56  	}
57  }
