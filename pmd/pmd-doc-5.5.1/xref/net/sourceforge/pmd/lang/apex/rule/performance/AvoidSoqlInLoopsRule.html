4   package net.sourceforge.pmd.lang.apex.rule.performance;
5   
6   import net.sourceforge.pmd.lang.ast.Node;
7   import net.sourceforge.pmd.lang.apex.ast.ASTSoqlExpression;
8   import net.sourceforge.pmd.lang.apex.ast.ASTDoLoopStatement;
9   import net.sourceforge.pmd.lang.apex.ast.ASTForLoopStatement;
10  import net.sourceforge.pmd.lang.apex.ast.ASTForEachStatement;
11  import net.sourceforge.pmd.lang.apex.ast.ASTReturnStatement;
12  import net.sourceforge.pmd.lang.apex.ast.ASTWhileLoopStatement;
13  import net.sourceforge.pmd.lang.apex.rule.AbstractApexRule;
14  
15  public class AvoidSoqlInLoopsRule extends AbstractApexRule {
16  
17  	public AvoidSoqlInLoopsRule() {
18  		setProperty(CODECLIMATE_CATEGORIES, new String[]{ "Performance" });
19  		// Note: Often more complicated as just moving the SOQL a few lines. Involves Maps...
20  		setProperty(CODECLIMATE_REMEDIATION_MULTIPLIER, 150);
21  		setProperty(CODECLIMATE_BLOCK_HIGHLIGHTING, false);
22  	}
23  
24  	@Override
25  	public Object visit(ASTSoqlExpression node, Object data) {
26  		if (insideLoop(node) && parentNotReturn(node) && parentNotForEach(node)) {
27  			addViolation(data, node);
28  		}
29  		return data;
30  	}
31  
32  	private boolean parentNotReturn(ASTSoqlExpression node) {
33  		return !(node.jjtGetParent() instanceof ASTReturnStatement);
34  	}
35  
36  	private boolean parentNotForEach(ASTSoqlExpression node) {
37  		return !(node.jjtGetParent() instanceof ASTForEachStatement);
38  	}
39  
40  	private boolean insideLoop(ASTSoqlExpression node) {
41  		Node n = node.jjtGetParent();
42  
43  		while (n != null) {
44  			if (n instanceof ASTDoLoopStatement || n instanceof ASTWhileLoopStatement
45  					|| n instanceof ASTForLoopStatement || n instanceof ASTForEachStatement) {
46  				return true;
47  			}
48  			n = n.jjtGetParent();
49  		}
50  
51  		return false;
52  	}
53  }
