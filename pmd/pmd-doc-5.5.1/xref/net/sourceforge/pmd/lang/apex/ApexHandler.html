10  import net.sourceforge.pmd.lang.Parser;

13  import net.sourceforge.pmd.lang.XPathHandler;

18  import net.sourceforge.pmd.lang.ast.xpath.AbstractASTXPathHandler;
19  import net.sourceforge.pmd.lang.rule.RuleViolationFactory;
20  
21  public class ApexHandler extends AbstractLanguageVersionHandler {
22  
23  	@Override
24  	public XPathHandler getXPathHandler() {
25  		return new AbstractASTXPathHandler() {
26  			public void initialize() {
27  			}
28  
29  			public void initialize(IndependentContext context) {
30  			}
31  		};
32  	}
33  
34  	public RuleViolationFactory getRuleViolationFactory() {
35  		return ApexRuleViolationFactory.INSTANCE;
36  	}
37  
38  	@Override
39  	public ParserOptions getDefaultParserOptions() {
40  		return new ApexParserOptions();
41  	}
42  
43  	public Parser getParser(ParserOptions parserOptions) {
44  		return new ApexParser(parserOptions);
45  	}
46  
47  	@Override
48  	public VisitorStarter getDumpFacade(Writer writer, String prefix, boolean recurse) {
49  		return new VisitorStarter() {
50  			public void start(Node rootNode) {
51  				new DumpFacade().initializeWith(writer, prefix, recurse, (ApexNode<?>) rootNode);
52  			}
53  		};
54  	}
55  }
