34   * @author Romain Pelisse, bugfix for [ 1522517 ] False +: UselessOverridingMethod
35   */
36  public class UselessOverridingMethodRule extends AbstractJavaRule {
37      private final List<String> exceptions;
38      private boolean ignoreAnnotations;
39      private static final String CLONE = "clone";
40      private static final String OBJECT = "Object";
41  
42      private static final BooleanProperty IGNORE_ANNOTATIONS_DESCRIPTOR = new BooleanProperty(
43                  "ignoreAnnotations", "Ignore annotations", false, 1.0f);
44  
45      public UselessOverridingMethodRule() {
46          definePropertyDescriptor(IGNORE_ANNOTATIONS_DESCRIPTOR);
47  
48          exceptions = new ArrayList<>(1);
49          exceptions.add("CloneNotSupportedException");
50      }
51  
52      @Override
53      public Object visit(ASTCompilationUnit node, Object data) {
54          init();
55          return super.visit(node, data);
56      }
57  
58      private void init() {
59          ignoreAnnotations = getProperty(IGNORE_ANNOTATIONS_DESCRIPTOR);
60      }

62      @Override
63      public Object visit(ASTClassOrInterfaceDeclaration clz, Object data) {
64          if (clz.isInterface()) {
65              return data;
66          }
67          return super.visit(clz, data);
68      }

70      //TODO: this method should be externalize into an utility class, shouldn't it ?
71      private boolean isMethodType(ASTMethodDeclaration node, String methodType) {
72          boolean result = false;
73          ASTResultType type = node.getResultType();
74          if (type != null) {
75              result = type.hasDescendantMatchingXPath("./Type/ReferenceType/ClassOrInterfaceType[@Image = '"
76                          + methodType + "']");
77          }
78          return result;
79      }
80  
81      //TODO: this method should be externalize into an utility class, shouldn't it ?
82      private boolean isMethodThrowingType(ASTMethodDeclaration node, List<String> exceptedExceptions) {
83          boolean result = false;
84          ASTNameList thrownsExceptions = node.getFirstChildOfType(ASTNameList.class);
85          if (thrownsExceptions != null) {
86              List<ASTName> names = thrownsExceptions.findChildrenOfType(ASTName.class);
87              for (ASTName name : names) {
88                  for (String exceptedException : exceptedExceptions) {
89                      if (exceptedException.equals(name.getImage())) {
90                          result = true;
91                      }
92                  }
93              }
94          }
95          return result;
96      }

98      private boolean hasArguments(ASTMethodDeclaration node) {
99          return node.hasDescendantMatchingXPath("./MethodDeclarator/FormalParameters/*");
100     }

102     @Override
103     public Object visit(ASTMethodDeclaration node, Object data) {
104         // Can skip abstract methods and methods whose only purpose is to
105         // guarantee that the inherited method is not changed by finalizing
106         // them.
107         if (node.isAbstract() || node.isFinal() || node.isNative() || node.isSynchronized()) {
108             return super.visit(node, data);
109         }
110         // We can also skip the 'clone' method as they are generally
111         // 'useless' but as it is considered a 'good practice' to
112         // implement them anyway ( see bug 1522517)
113         if (CLONE.equals(node.getMethodName()) && node.isPublic() && !this.hasArguments(node)
114                 && this.isMethodType(node, OBJECT) && this.isMethodThrowingType(node, exceptions)) {
115             return super.visit(node, data);
116         }
117 
118         ASTBlock block = node.getBlock();
119         if (block == null) {
120             return super.visit(node, data);
121         }
122         //Only process functions with one BlockStatement
123         if (block.jjtGetNumChildren() != 1 || block.findDescendantsOfType(ASTStatement.class).size() != 1) {
124             return super.visit(node, data);
125         }
126 
127         Node statement = block.jjtGetChild(0).jjtGetChild(0);
128         if (statement.jjtGetChild(0).jjtGetNumChildren() == 0) {
129             return data; // skips empty return statements
130         }
131         Node statementGrandChild = statement.jjtGetChild(0).jjtGetChild(0);
132         ASTPrimaryExpression primaryExpression;
133 
134         if (statementGrandChild instanceof ASTPrimaryExpression) {
135             primaryExpression = (ASTPrimaryExpression) statementGrandChild;
136         } else {
137             List<ASTPrimaryExpression> primaryExpressions = findFirstDegreeChildrenOfType(statementGrandChild,
138                     ASTPrimaryExpression.class);
139             if (primaryExpressions.size() != 1) {
140                 return super.visit(node, data);
141             }
142             primaryExpression = primaryExpressions.get(0);
143         }
144 
145         ASTPrimaryPrefix primaryPrefix = findFirstDegreeChildrenOfType(primaryExpression, ASTPrimaryPrefix.class)
146                 .get(0);
147         if (!primaryPrefix.usesSuperModifier()) {
148             return super.visit(node, data);
149         }
150 
151         List<ASTPrimarySuffix> primarySuffixList = findFirstDegreeChildrenOfType(primaryExpression,
152                 ASTPrimarySuffix.class);
153         if (primarySuffixList.size() != 2) {
154             // extra method call on result of super method
155             return super.visit(node, data);
156         }
157 
158         ASTMethodDeclarator methodDeclarator = findFirstDegreeChildrenOfType(node, ASTMethodDeclarator.class).get(0);
159         ASTPrimarySuffix primarySuffix = primarySuffixList.get(0);
160         if (!primarySuffix.hasImageEqualTo(methodDeclarator.getImage())) {
161         	return super.visit(node, data);
162         }
163         //Process arguments
164         primarySuffix = primarySuffixList.get(1);
165         ASTArguments arguments = (ASTArguments) primarySuffix.jjtGetChild(0);
166         ASTFormalParameters formalParameters = (ASTFormalParameters) methodDeclarator.jjtGetChild(0);
167         if (formalParameters.jjtGetNumChildren() != arguments.jjtGetNumChildren()) {
168             return super.visit(node, data);
169         }
170 
171         if (!ignoreAnnotations) {
172             ASTClassOrInterfaceBodyDeclaration parent = (ASTClassOrInterfaceBodyDeclaration) node.jjtGetParent();
173             for (int i = 0; i < parent.jjtGetNumChildren(); i++) {
174                 Node n = parent.jjtGetChild(i);
175                 if (n instanceof ASTAnnotation) {
176                     if (n.jjtGetChild(0) instanceof ASTMarkerAnnotation) {
177                         // @Override is ignored
178                         if ("Override".equals(((ASTName) n.jjtGetChild(0).jjtGetChild(0)).getImage())) {
179                             continue;
180                         }
181                     }
182                     return super.visit(node, data);
183                 }
184             }
185         }
186 
187         if (arguments.jjtGetNumChildren() == 0) {
188             addViolation(data, node, getMessage());
189         } else {
190             ASTArgumentList argumentList = (ASTArgumentList) arguments.jjtGetChild(0);
191             for (int i = 0; i < argumentList.jjtGetNumChildren(); i++) {
192                 Node expressionChild = argumentList.jjtGetChild(i).jjtGetChild(0);
193                 if (!(expressionChild instanceof ASTPrimaryExpression) || expressionChild.jjtGetNumChildren() != 1) {
194                     return super.visit(node, data); //The arguments are not simply passed through
195                 }
196 
197                 ASTPrimaryExpression argumentPrimaryExpression = (ASTPrimaryExpression) expressionChild;
198                 ASTPrimaryPrefix argumentPrimaryPrefix = (ASTPrimaryPrefix) argumentPrimaryExpression.jjtGetChild(0);
199                 if (argumentPrimaryPrefix.jjtGetNumChildren() == 0) {
200                     return super.visit(node, data); //The arguments are not simply passed through (using "this" for instance)
201                 }
202                 Node argumentPrimaryPrefixChild = argumentPrimaryPrefix.jjtGetChild(0);
203                 if (!(argumentPrimaryPrefixChild instanceof ASTName)) {
204                     return super.visit(node, data); //The arguments are not simply passed through
205                 }
206 
207                 if (formalParameters.jjtGetNumChildren() < i + 1) {
208                     return super.visit(node, data); // different number of args
209                 }
210 
211                 ASTName argumentName = (ASTName) argumentPrimaryPrefixChild;
212                 ASTFormalParameter formalParameter = (ASTFormalParameter) formalParameters.jjtGetChild(i);
213                 ASTVariableDeclaratorId variableId = findFirstDegreeChildrenOfType(formalParameter,
214                         ASTVariableDeclaratorId.class).get(0);
215                 if (!argumentName.hasImageEqualTo(variableId.getImage())) {
216                     return super.visit(node, data); //The arguments are not simply passed through
217                 }
218 
219             }
220             addViolation(data, node, getMessage()); //All arguments are passed through directly
221         }
222         return super.visit(node, data);
223     }

225     public <T> List<T> findFirstDegreeChildrenOfType(Node n, Class<T> targetType) {
226         List<T> l = new ArrayList<>();
227         lclFindChildrenOfType(n, targetType, l);
228         return l;
229     }
230 
231     private <T> void lclFindChildrenOfType(Node node, Class<T> targetType, List<T> results) {
232         if (node.getClass().equals(targetType)) {
233             results.add((T) node);
234         }
235 
236         if (node instanceof ASTClassOrInterfaceDeclaration && ((ASTClassOrInterfaceDeclaration) node).isNested()) {
237             return;
238         }
239 
240         if (node instanceof ASTClassOrInterfaceBodyDeclaration
241                 && ((ASTClassOrInterfaceBodyDeclaration) node).isAnonymousInnerClass()) {
242             return;
243         }
244 
245         for (int i = 0; i < node.jjtGetNumChildren(); i++) {
246             Node child = node.jjtGetChild(i);
247             if (child.getClass().equals(targetType)) {
248                 results.add((T) child);
249             }
250         }
251     }
