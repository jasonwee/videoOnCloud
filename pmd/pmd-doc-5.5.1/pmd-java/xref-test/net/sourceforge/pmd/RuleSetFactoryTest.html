11  /**
12   * Test java's rulesets
13   */
14  public class RuleSetFactoryTest extends AbstractRuleSetFactoryTest {
15  
16      @Test
17      public void testExclusionOfUselessParantheses() throws RuleSetNotFoundException {
18          RuleSetReferenceId ref = createRuleSetReferenceId("<?xml version=\"1.0\"?>\n" +
19                  "<ruleset name=\"Custom ruleset for tests\"\n" +
20                  "    xmlns=\"http://pmd.sourceforge.net/ruleset/2.0.0\"\n" +
21                  "    xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\"\n" +
22                  "    xsi:schemaLocation=\"http://pmd.sourceforge.net/ruleset/2.0.0 http://pmd.sourceforge.net/ruleset_2_0_0.xsd\">\n" +
23                  "  <description>Custom ruleset for tests</description>\n" +
24                  "  <rule ref=\"rulesets/java/unnecessary.xml\">\n" +
25                  "    <exclude name=\"UselessParentheses\"/>\n" +
26                  "  </rule>\n" +
27                  "</ruleset>\n");
28          RuleSetFactory ruleSetFactory = new RuleSetFactory();
29          RuleSet ruleset = ruleSetFactory.createRuleSet(ref);
30          Rule rule = ruleset.getRuleByName("UselessParentheses");
31          assertNull(rule);
32      }
33  
34      /**
35       * Makes sure that the internal dogfood.xml ruleset is valid and doesn't reference any
36       * unknown rules.
37       * @throws RuleSetNotFoundException if dogfood couldn't be found at all
38       */
39      @Test
40      public void testDogfoodRuleset() throws RuleSetNotFoundException {
41          RuleSetReferenceId ref = RuleSetReferenceId.parse("rulesets/internal/dogfood.xml").get(0);
42          RuleSetFactory ruleSetFactory = new RuleSetFactory();
43          RuleSet ruleset = ruleSetFactory.createRuleSet(ref);
44          assertNotNull(ruleset);
45      }
