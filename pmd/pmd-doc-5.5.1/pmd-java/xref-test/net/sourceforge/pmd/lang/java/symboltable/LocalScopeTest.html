24      @Test
25      public void testNameWithThisOrSuperIsNotFlaggedAsUnused() {
26          LocalScope scope = new LocalScope();
27          ASTName name = new ASTName(1);
28          name.setImage("foo");
29          ASTPrimaryPrefix prefix = new ASTPrimaryPrefix(2);
30          prefix.setUsesThisModifier();
31          name.jjtAddChild(prefix, 1);
32          JavaNameOccurrence occ = new JavaNameOccurrence(name, "foo");
33          scope.addNameOccurrence(occ);
34          assertFalse(scope.getDeclarations().keySet().iterator().hasNext());
35      }

50      @Test
51      public void testLocalVariableDeclarationFound() {
52          parseCode(TEST1);
53          List<ASTVariableDeclaratorId> nodes = acu.findDescendantsOfType(ASTVariableDeclaratorId.class);
54          ASTVariableDeclaratorId node = nodes.get(0);
55          Map<NameDeclaration, List<NameOccurrence>> vars = node.getScope().getDeclarations();
