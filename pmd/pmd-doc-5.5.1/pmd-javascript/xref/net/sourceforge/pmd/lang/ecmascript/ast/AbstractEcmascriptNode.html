4   package net.sourceforge.pmd.lang.ecmascript.ast;
5   
6   import org.mozilla.javascript.ast.AstNode;
7   
8   import net.sourceforge.pmd.lang.ast.AbstractNode;
9   import net.sourceforge.pmd.lang.ast.SourceCodePositioner;
10  
11  public abstract class AbstractEcmascriptNode<T extends AstNode> extends AbstractNode implements EcmascriptNode<T> {
12  
13      protected final T node;
14  
15      public AbstractEcmascriptNode(T node) {
16  	super(node.getType());
17  	this.node = node;
18      }
19  
20      /* package private */
21      void calculateLineNumbers(SourceCodePositioner positioner) {
22  	int startOffset = node.getAbsolutePosition();
23  	int endOffset = startOffset + node.getLength();
24  
25  	this.beginLine = positioner.lineNumberFromOffset(startOffset);
26  	this.beginColumn = positioner.columnFromOffset(this.beginLine, startOffset);
27  	this.endLine = positioner.lineNumberFromOffset(endOffset);
28  	this.endColumn = positioner.columnFromOffset(this.endLine, endOffset) - 1; // end column is inclusive
29  	if (this.endColumn < 0) {
30  	    this.endColumn = 0;
31  	}
32      }
33  
34      /**
35       * Accept the visitor. *
36       */
37      public Object jjtAccept(EcmascriptParserVisitor visitor, Object data) {
38  	return visitor.visit(this, data);
39      }
40  
41      /**
42       * Accept the visitor. *
43       */
44      public Object childrenAccept(EcmascriptParserVisitor visitor, Object data) {
45  	if (children != null) {
46  	    for (int i = 0; i < children.length; ++i) {
47  		@SuppressWarnings("unchecked") // we know that the children here are all EcmascriptNodes
48  		EcmascriptNode<T> ecmascriptNode = (EcmascriptNode<T>) children[i];
49  		ecmascriptNode.jjtAccept(visitor, data);
50  	    }
51  	}
52  	return data;
53      }
54      
55      public T getNode() {
56  	return node;
57      }
58  
59      public String getJsDoc() {
60  	return node.getJsDoc();
61      }
62  
63      public boolean hasSideEffects() {
64  	return node.hasSideEffects();
65      }
66  
67      @Override
68      public String toString() {
69  	return node.shortName();
70      }
71  }
