6   import net.sourceforge.pmd.lang.rule.properties.StringMultiProperty;
7   import net.sourceforge.pmd.lang.rule.properties.StringProperty;

9   /**
10   * Sample rule that detect any node with an image of "Foo", similar to {@link FooRule}.
11   * It additionally has some properties in order to test the renderers.
12   * Used for testing.
13   */
14  public class RuleWithProperties extends FooRule {
15      public static StringProperty STRING_PROPERTY_DESCRIPTOR = new StringProperty("stringProperty", "simple string property", null, 1.0f);
16      public static StringMultiProperty MULTI_STRING_PROPERTY_DESCRIPTOR =
17              new StringMultiProperty("multiString", "multi string property", new String[]{"default1", "default2"}, 2.0f, ',');
18  
19      public RuleWithProperties() {
20          definePropertyDescriptor(STRING_PROPERTY_DESCRIPTOR);
21          definePropertyDescriptor(MULTI_STRING_PROPERTY_DESCRIPTOR);
22      }
23  }
