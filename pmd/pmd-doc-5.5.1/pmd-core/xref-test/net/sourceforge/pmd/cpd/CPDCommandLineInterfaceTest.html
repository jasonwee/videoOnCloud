10  import org.junit.contrib.java.lang.system.LogMode;

15  import net.sourceforge.pmd.PMD;

17  public class CPDCommandLineInterfaceTest {
18      @Rule
19      public final TestRule restoreSystemProperties = new RestoreSystemProperties();
20      @Rule
21      public final StandardOutputStreamLog log = new StandardOutputStreamLog(LogMode.LOG_ONLY);
22  
23      @Test
24      public void testEmptyResultRendering() {
25          System.setProperty(CPDCommandLineInterface.NO_EXIT_AFTER_RUN, "true");
26          CPDCommandLineInterface.main(new String[] {
27                  "--minimum-tokens", "340",
28                  "--language", "java",
29                  "--files", "src/test/resources/net/sourceforge/pmd/cpd/files/",
30                  "--format", "xml"
31          });
32          Assert.assertEquals("<?xml version=\"1.0\" encoding=\"UTF-8\"?>" + "\n" + "<pmd-cpd/>" + PMD.EOL, log.getLog());
33      }
