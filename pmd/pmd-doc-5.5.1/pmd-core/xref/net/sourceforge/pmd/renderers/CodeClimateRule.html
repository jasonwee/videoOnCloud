4   package net.sourceforge.pmd.renderers;
5   
6   import static net.sourceforge.pmd.renderers.CodeClimateRenderer.REMEDIATION_POINTS_DEFAULT;
7   
8   import net.sourceforge.pmd.Rule;
9   import net.sourceforge.pmd.lang.rule.properties.EnumeratedMultiProperty;
10  import net.sourceforge.pmd.lang.rule.properties.IntegerProperty;
11  import net.sourceforge.pmd.lang.rule.properties.BooleanProperty;
12  
13  /**
14   * This interface tags a Rule specifying properties required for the Code
15   * Climate output format. Each rule implementing this interface must
16   * define the two code climate properties "categories" and "remediation multiplier".
17   */
18  public interface CodeClimateRule extends Rule {
19      /**
20       * Defines the code climate categories for which this rule will find violations.
21       * Possible categories are: Bug Risk, Clarity, Compatibility, Complexity, Duplication,
22       * Performance, Security, Style.
23       *
24       * @see <a href="https://github.com/codeclimate/spec/blob/master/SPEC.md#categories">Code Climate Spec</a>
25       */
26      EnumeratedMultiProperty<String> CODECLIMATE_CATEGORIES = new EnumeratedMultiProperty<String>(
27              "cc_categories", "Code Climate Categories",
28              new String[] { "Bug Risk", "Clarity", "Compatibility", "Complexity", "Duplication", "Performance", "Security", "Style" },
29              new String[] { "Bug Risk", "Clarity", "Compatibility", "Complexity", "Duplication", "Performance", "Security", "Style" },
30              new int[] { 7 }, 1.0f);
31  
32      /**
33       * Defines the remediation points for this rule. The remediation points are not set directly but are expressed as
34       * a multiplier. The {@link CodeClimateRenderer} takes this and multiplies it with {@link CodeClimateRenderer#REMEDIATION_POINTS_DEFAULT},
35       * which is the baseline points for a trivial fix (value is 50000).
36       *
37       * @see CodeClimateRenderer#REMEDIATION_POINTS_DEFAULT
38       */
39      // Note: We use a multiplier to the Code Climate default of 50000 for the
40      // simplest possible remediation
41      IntegerProperty CODECLIMATE_REMEDIATION_MULTIPLIER = new IntegerProperty("cc_remediation_points_multiplier",
42              "Code Climate Remediation Points multiplier", Integer.MIN_VALUE,
43              Integer.MAX_VALUE / REMEDIATION_POINTS_DEFAULT, 1, 1.0f);
44      
45      /**
46       * Defines if the whole related block or just the first line of the issue should be highlighted at the Code Climate Platform.
47       * By default the block highlighting is disabled for reasons of clarity.
48       */
49      BooleanProperty CODECLIMATE_BLOCK_HIGHLIGHTING = new BooleanProperty("cc_block_highlighting", 
50      		"Code Climate Block Highlighting", false, 1.0f);
51  }
