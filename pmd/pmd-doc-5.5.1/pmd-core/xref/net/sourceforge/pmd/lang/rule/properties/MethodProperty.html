15  /**
16   * Defines a property type that can specify a single method to use as part of a
17   * rule.
18   *
19   * Rule developers can limit the rules to those within designated packages per
20   * the 'legalPackages' argument in the constructor which can be an array of
21   * partial package names, i.e., ["java.lang", "com.mycompany" ].
22   *
23   * @author Brian Remedios
24   */
25  public class MethodProperty extends AbstractPackagedProperty<Method> {
26  
27      public static final char CLASS_METHOD_DELIMITER = '#';
28      public static final char METHOD_ARG_DELIMITER = ',';
29      public static final char[] METHOD_GROUP_DELIMITERS = new char[] { '(', ')' };
30  
31      private static final String ARRAY_FLAG = "[]";
32      private static final Map<Class<?>, String> TYPE_SHORTCUTS = ClassUtil.getClassShortNames();
33  
34      public static final PropertyDescriptorFactory FACTORY = new BasicPropertyDescriptorFactory<MethodProperty>(
35              Method.class, PACKAGED_FIELD_TYPES_BY_KEY) {
36  
37          @Override
38          public MethodProperty createWith(Map<String, String> valuesById) {
39              char delimiter = delimiterIn(valuesById);
40              return new MethodProperty(nameIn(valuesById), descriptionIn(valuesById), defaultValueIn(valuesById),
41                      legalPackageNamesIn(valuesById, delimiter), 0f);
42          }
43      };

90      /**
91       * @param cls Class<?>
92       * @return String
93       */
94      private static String shortestNameFor(Class<?> cls) {
95          String compactName = TYPE_SHORTCUTS.get(cls);
96          return compactName == null ? cls.getName() : compactName;
97      }

99      /**
100      * Return the value of `method' as a string that can be easily recognized
101      * and parsed when we see it again.
102      *
103      * @param method the method to convert
104      * @return the string value
105      */
106     public static String asStringFor(Method method) {
107         StringBuilder sb = new StringBuilder();
108         asStringOn(method, sb);
109         return sb.toString();
110     }

120     /**
121      * @param type Class<?>
122      * @param sb StringBuilder
123      */
124     private static void serializedTypeIdOn(Class<?> type, StringBuilder sb) {
125 
126         Class<?> arrayType = type.getComponentType();
127         if (arrayType == null) {
128             sb.append(shortestNameFor(type));
129             return;
130         }
131         sb.append(shortestNameFor(arrayType)).append(ARRAY_FLAG);
132     }
133 
134     /**
135      * Serializes the method signature onto the specified buffer.
136      *
137      * @param method Method
138      * @param sb StringBuilder
139      */
140     public static void asStringOn(Method method, StringBuilder sb) {
141 
142         Class<?> clazz = method.getDeclaringClass();
143 
144         sb.append(shortestNameFor(clazz));
145         sb.append(CLASS_METHOD_DELIMITER);
146         sb.append(method.getName());
147 
148         sb.append(METHOD_GROUP_DELIMITERS[0]);
149 
150         Class<?>[] argTypes = method.getParameterTypes();
151         if (argTypes.length == 0) {
152             sb.append(METHOD_GROUP_DELIMITERS[1]);
153             return;
154         }
155 
156         serializedTypeIdOn(argTypes[0], sb);
157         for (int i = 1; i < argTypes.length; i++) {
158             sb.append(METHOD_ARG_DELIMITER);
159             serializedTypeIdOn(argTypes[i], sb);
160         }
161         sb.append(METHOD_GROUP_DELIMITERS[1]);
162     }
163 
164     /**
165      * @param typeName String
166      * @return Class<?>
167      */
168     private static Class<?> typeFor(String typeName) {
169 
170         Class<?> type = null;
171 
172         if (typeName.endsWith(ARRAY_FLAG)) {
173             String arrayTypeName = typeName.substring(0, typeName.length() - ARRAY_FLAG.length());
174             type = typeFor(arrayTypeName); // recurse
175             return Array.newInstance(type, 0).getClass(); // TODO is there a
176                                                           // better way to get
177                                                           // an array type?
178         }
179 
180         type = ClassUtil.getTypeFor(typeName); // try shortcut first
181         if (type != null) {
182             return type;
183         }
184 
185         try {
186             return Class.forName(typeName);
187         } catch (Exception ex) {
188             return null;
189         }
190     }

192     /**
193      * Returns the method specified within the string argument after parsing out
194      * its source class and any optional arguments. Callers need to specify the
195      * delimiters expected between the various elements. I.e.:
196      *
197      * "String#isEmpty()" "String#indexOf(int)" "String#substring(int,int)"
198      *
199      * If a method isn't part of the specified class we will walk up any
200      * superclasses to Object to try and find it.
201      *
202      * If the classes are listed in the ClassUtil class within in Typemaps then
203      * you likely can avoid specifying fully-qualified class names per the above
204      * example.
205      *
206      * Returns null if a matching method cannot be found.
207      *
208      * @param methodNameAndArgTypes
209      * @param classMethodDelimiter
210      * @param methodArgDelimiter
211      * @return Method
212      */
213     public static Method methodFrom(String methodNameAndArgTypes, char classMethodDelimiter, char methodArgDelimiter) {
214 
215         // classname#methodname(arg1,arg2)
216         // 0 1 2
217 
218         int delimPos0 = -1;
219         if (methodNameAndArgTypes != null) {
220             delimPos0 = methodNameAndArgTypes.indexOf(classMethodDelimiter);
221         } else {
222             return null;
223         }
224 
225         if (delimPos0 < 0) {
226             return null;
227         }
228 
229         String className = methodNameAndArgTypes.substring(0, delimPos0);
230         Class<?> type = ClassUtil.getTypeFor(className);
231         if (type == null) {
232             return null;
233         }
234 
235         int delimPos1 = methodNameAndArgTypes.indexOf(METHOD_GROUP_DELIMITERS[0]);
236         if (delimPos1 < 0) {
237             String methodName = methodNameAndArgTypes.substring(delimPos0 + 1);
238             return ClassUtil.methodFor(type, methodName, ClassUtil.EMPTY_CLASS_ARRAY);
239         }
240 
241         String methodName = methodNameAndArgTypes.substring(delimPos0 + 1, delimPos1);
242         if (StringUtil.isEmpty(methodName)) {
243             return null;
244         } // missing method name?
245 
246         int delimPos2 = methodNameAndArgTypes.indexOf(METHOD_GROUP_DELIMITERS[1]);
247         if (delimPos2 < 0) {
248             return null;
249         } // error!
250 
251         String argTypesStr = methodNameAndArgTypes.substring(delimPos1 + 1, delimPos2);
252         if (StringUtil.isEmpty(argTypesStr)) {
253             return ClassUtil.methodFor(type, methodName, ClassUtil.EMPTY_CLASS_ARRAY);
254         } // no arg(s)
255 
256         String[] argTypeNames = StringUtil.substringsOf(argTypesStr, methodArgDelimiter);
257         Class<?>[] argTypes = new Class[argTypeNames.length];
258         for (int i = 0; i < argTypes.length; i++) {
259             argTypes[i] = typeFor(argTypeNames[i]);
260         }
261 
262         return ClassUtil.methodFor(type, methodName, argTypes);
263     }

267      * @return Method
268      */

270         return methodFrom(methodStr, CLASS_METHOD_DELIMITER, METHOD_ARG_DELIMITER);
271     }

289     @Override
290     protected String packageNameOf(Object item) {
291 
292         final Method method = (Method) item;
293         return method.getDeclaringClass().getName() + '.' + method.getName();
